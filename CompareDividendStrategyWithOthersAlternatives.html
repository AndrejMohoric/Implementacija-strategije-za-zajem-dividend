def ReadFile(path):
    data_list = [[],[]]
    data = open(path, "r")
    counter = 0
    stevecVrstic = -1
    for line in data:
        stevecVrstic = stevecVrstic + 1 
        if counter == 0:
            counter = 1
        else:
            x = line.split(" ") #VRSTICA
            stevec = -1
            for i in range(len(x)):
                if (stevec ==3 ):
                    stevec = -1 
                if(x[i]!= ""):
                    #print(stevec)
                    if(stevec == 0):#DATE
                        #print(x[i].rstrip())
                        data_list[stevec].append(x[i].rstrip())
                    if(stevec == 1):#Price
                        #print(x[i].rstrip())
                        data_list[stevec].append(x[i].rstrip())
                    stevec= stevec + 1
    return data_list

def sharpe_ratio(return_series, N, rf):
    if(return_series.name == 'BTC'):
        N = 365
        mean = return_series.mean() * N -rf
        sigma = return_series.std() * np.sqrt(N)
    else:
        mean = return_series.mean() * N -rf
        sigma = return_series.std() * np.sqrt(N)
    return mean / sigma

## Sortino Ratio
######################################
def sortino_ratio(series, N,rf):
    mean = series.mean() * N -rf
    std_neg = series[series<0].std()*np.sqrt(N)
    return mean/std_neg

######################################
## Max Drawdown
######################################
def max_drawdown(return_series):
    comp_ret = (return_series+1).cumprod()
    peak = comp_ret.expanding(min_periods=1).max()
    dd = (comp_ret/peak)-1
    return dd.min()


max_drawdowns = df.apply(max_drawdown,axis=0)
